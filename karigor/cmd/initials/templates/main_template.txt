import uvicorn
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI(
    title="Karigor Enterprise API",
    description="Automated Feature-Driven Backend for 100+ Developers",
    version="1.0.0"
)

# CORS Middleware Configuration
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.get("/health", tags=["System"])
async def health_check():
    return {"status": "healthy", "engine": "Karigor CLI"}

def main():
    print("🚀 Starting Karigor App via Uvicorn...")
    uvicorn.run(app, host="127.0.0.1", port=8000)

if __name__ == '__main__':
    main()
